// Problem: A. Little Elephant and Function
// Contest: Codeforces - Codeforces Round #136 (Div. 2)
// URL: https://codeforces.com/problemset/problem/221/A
// Memory Limit: 256 MB
// Time Limit: 2000 ms

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    cout << n;
    for (int i = 1; i <= n - 1; ++i)
        cout << ' ' << i;

    return 0;
}
